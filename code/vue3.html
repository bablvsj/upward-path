1.vite.config.ts 配置解析 ​

ts
// vite.config.js
import vue from "@vitejs/plugin-vue";
import { resolve } from "path";
import { defineConfig, loadEnv, ConfigEnv } from "vite";
import viteCompression from "vite-plugin-compression";
import AutoImport from "unplugin-auto-import/vite";  // https://github.com/unplugin/unplugin-auto-import
import Components from "unplugin-vue-components/vite";
import { ElementPlusResolver } from "unplugin-vue-components/resolvers";

const pathResolve = (dir: string) => {
  return resolve(__dirname, ".", dir);
};

const alias: Record<string, string> = {
  "@": pathResolve("./src/"),
};

const viteConfig = defineConfig((mode: ConfigEnv) => {
  return {
    plugins: [
      vue(),
      viteCompression(),
      // 相关方法存储在auto-import.d.ts文件中
      AutoImport({
        dts : true, // 生成dts文件
        // 自动导入Vue 相关函数，如：ref, reactive, toRef等
        imports: ["vue", "vue-router"],
        //自动导入 Element Plus 相关函数，如：ElMessage, ElMessageBox... (带样式)
        resolvers: [ElementPlusResolver()],
      }),
      // 组件存储在components.d.ts文件中
      Components({
        // 自动导入 Element Plus 组件
        resolvers: [ElementPlusResolver()],
      }),
    ],
    resolve: { alias },
    // base: mode.command === 'serve' ? './' : env.VITE_PUBLIC_PATH,
    base: "./",
    server: {
      //  本地服务器
      host: "0.0.0.0",
      port: 8086,
      oper: true,
      hmr: true,
      proxy: {
        "/api": {
          target: "http://10.52.18.8:6580",
          changeOrigin: true,
          rewrite: (path) => path.replace(/^\/api/, "/api"),
        },
      },
    },
    build: {
      // 打包
      outDir: "dist",
      chunkSizeWarningLimit: 1500,
      sourcemap: false,
      productionSourceMap: false,
      cssCodeSplit: false,
      rollupOptions: {
        output: {
          chunkFileNames: "assets/js/[name]-[hash].js",
          entryFileNames: "assets/js/[name]-[hash].js",
          assetFileNames: "assets/[ext]/[name]-[hash].[ext]",
          manualChunks(id) {
            if (id.includes("node_modules")) {
              return id
                .toString()
                .split("node_modules/")[1]
                .split("/")[0]
                .toString();
            }
          },
        },
      },
    },
    css: { preprocessorOptions: { css: { charset: false } } },
  };
});
export default viteConfig;
