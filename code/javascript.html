2. JSBridge ​

js
//主要用来和原生的app交互  在H5的项目中调用原生的方法  主要是接收原生的参数的
let udd = navigator.userAgent; // 获取浏览器的信息 类型以及系统
let isaddAndroid = udd.indexOf("Android") > -1 || udd.indexOf("Adr") > -1; // 安卓的终端
let ioadds = !!navigator.userAgent.match(/\(i[^;]+;( U;)? CPU.+Mac OS X/); // ioadds的终端
// 安卓
function setupWebViewJavascriptBridge(callback) {
  if (isaddAndroid) {
    // console.log("进的是isaddAndroid")
    if (window.WebViewJavascriptBridge) {
      callback(WebViewJavascriptBridge);
    } else {
      document.addEventListener("WebViewJavascriptBridgeReady",
        function() {
          callback(WebViewJavascriptBridge);
        }, false);
    }
    //sessionStorage.phoneType = "android";
  }
  // ioadds
  if (ioadds) {
    // console.log("进的是ioadds")
    if (window.WebViewJavascriptBridge) {
      return callback(WebViewJavascriptBridge);
    }
    if (window.WVJBCallbacks) {
      return window.WVJBCallbacks.push(callback);
    }
    window.WVJBCallbacks = [ callback ];
    let WVJBIframe = document.createElement("iframe");
    WVJBIframe.style.display = "none";
    WVJBIframe.src = 'wvjbscheme://__BRIDGE_LOADED__';
    // WVJBIframe.src = "wvjbscheme://__BRIDGE_LOADED__"; //这个是老的
    // WVJBIframe.src = "https://__bridge_loaded__"; // 这个方法会随时更新的
    // 到时候百度的时候注意一下copy最新的代码
    document.documentElement.appendChild(WVJBIframe);
    setTimeout(function() {
      document.documentElement.removeChild(WVJBIframe);
    }, 0);
    //console.log("tag", "ioadds");
    sessionStorage.phoneType = "ioadds";
  }
}

// 注册回调函数 第一次连接的时候调用 初始化函数安卓需要初始化 ioadds不需要
setupWebViewJavascriptBridge(function(bridge) {
  // 初始化
  if (isaddAndroid) {
    bridge.init(function(message, responseCallback) {
      let data = {
        "Javascript Response" : "wee!",
      };
      responseCallback(data);
    });
  }

});

function getBridge(funName, dataJson, callback) {
  setupWebViewJavascriptBridge(function(bridge) {
    bridge.callHandler(funName,{"handlerName": funName}, function(response) {
      callback && callback(response);
    });
  });
}

export { getBridge };
